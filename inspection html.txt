<script>
let originalCards = [];
let allLocsData = [];

window.addEventListener('load', function() {
  document.addEventListener("click", function(e) {
    if (!e.target.closest(".custom-select2-alternative")) {
      document.querySelectorAll(".dropdown-menu").forEach(m => m.style.display = "none");
    }
  });

  const locTrigger = document.getElementById("locTrigger");
  const locDropdownMenu = document.getElementById("locDropdownMenu");
  if (locTrigger) {
    locTrigger.addEventListener("click", function(e) {
      e.stopPropagation();
      const currentDisplay = locDropdownMenu.style.display;
      document.querySelectorAll(".dropdown-menu").forEach(m => m.style.display = "none");
      locDropdownMenu.style.display = currentDisplay === "block" ? "none" : "block";
    });
  }

  const locSearch = document.getElementById("locSearch");
  if (locSearch) {
    locSearch.addEventListener("input", function() {
      const val = this.value.toLowerCase();
      document.querySelectorAll("#locOptions .checkbox-item").forEach(item => {
        const text = item.textContent.toLowerCase();
        item.style.display = text.indexOf(val) > -1 ? "flex" : "none";
      });
    });
  }

  document.addEventListener("change", function(e) {
    if (e.target.matches("#locOptions input[type='checkbox']")) {
      updateTriggerText(document.getElementById("locTrigger"), document.getElementById("locOptions"), "Select Location");
      loadCardsFromServer();
    }
  });

  google.script.run.withSuccessHandler(function(initData) {
    if (initData) {
      allLocsData = initData.locations || [];
      const locOpts = document.getElementById("locOptions");
      renderCheckboxOptions(locOpts, allLocsData, [], "loc_chk");
      updateTriggerText(document.getElementById("locTrigger"), locOpts, "Select Location");
    }
  }).getInitData();
});

function updateTriggerText(triggerEl, optionsContainer, labelText) {
  if (!triggerEl || !optionsContainer) return;
  const checked = [];
  optionsContainer.querySelectorAll("input:checked").forEach(cb => checked.push(cb.value));
  if (checked.length === 0) {
    triggerEl.textContent = labelText; 
  } else if (checked.length <= 2) {
    triggerEl.textContent = checked.join(", ");
  } else {
    triggerEl.textContent = checked.length + " locations selected";
  }
}

function renderCheckboxOptions(container, items, checkedList, namePrefix) {
  if (!container) return;
  let html = "";
  const lowerChecked = checkedList.map(v => String(v).trim().toLowerCase());
  
  items.forEach((item, index) => {
    const isChecked = lowerChecked.includes(String(item).trim().toLowerCase()) ? "checked" : "";
    const id = namePrefix + "_" + index + "_" + Date.now();
    html += `
      <div class="checkbox-item">
        <input type="checkbox" value="${item}" id="${id}" ${isChecked}>
        <label style="display:inline; font-size:26px; font-weight:normal; margin:0;" for="${id}">${item}</label>
      </div>
    `;
  });
  container.innerHTML = html;
}

function addNewLocTag() {
  const inp = document.getElementById("locSearch");
  const val = inp.value.trim();
  if (!val) return;
  
  if (!isNaN(val.charAt(0))) {
    alert("⚠️ Tên Room không được phép bắt đầu bằng ký tự số!");
    return;
  }

  if (!allLocsData.some(v => v.toLowerCase() === val.toLowerCase())) {
    allLocsData.push(val);
    allLocsData.sort((a,b) => a.localeCompare(b));
  }
  const currentChecked = [];
  document.querySelectorAll("#locOptions input:checked").forEach(cb => currentChecked.push(cb.value));
  if (!currentChecked.includes(val)) currentChecked.push(val);
  
  renderCheckboxOptions(document.getElementById("locOptions"), allLocsData, currentChecked, "loc_chk");
  updateTriggerText(document.getElementById("locTrigger"), document.getElementById("locOptions"), "Select Location");
  inp.value = "";
  document.querySelectorAll("#locOptions .checkbox-item").forEach(item => item.style.display = "flex");
  loadCardsFromServer();
}

function loadCardsFromServer() {
  const selectedLocations = [];
  document.querySelectorAll("#locOptions input:checked").forEach(cb => selectedLocations.push(cb.value));

  if (selectedLocations.length === 0) {
    document.getElementById("cardsContainer").innerHTML = "";
    originalCards = [];
    return;
  }

  document.getElementById("cardsContainer").innerHTML = "<div style='font-size:26px;color:#666;padding:10px;'>⏳ Đang tải danh sách thẻ...</div>";

  google.script.run.withSuccessHandler(function(cards) {
    originalCards = cards || [];
    const container = document.getElementById("cardsContainer");
    container.innerHTML = "";
    originalCards.forEach(card => createCard(card));
  }).getCards(selectedLocations);
}
function createCard(card) {
  const container = document.getElementById("cardsContainer");
  const id = "card_" + Date.now() + "_" + Math.floor(Math.random() * 1000);
  const isNew = !card.number;

  const html = `
  <div class="card" data-id="${id}" data-isnew="${isNew}">
    <div class="card-title">${isNew ? "New Entry" : "Inspection Details"}</div>
    
    <div class="inspection-grid-row">
      <div class="grid-col-number">
        <label>Number</label>
        <input type="text" class="number form-control" 
               value="${card.number || ''}" ${isNew ? '' : 'readonly'} 
               placeholder="Số..." 
               oninput="handleNumberLookup(this, '${id}')" style="text-align: center;">
      </div>
      <div class="grid-col-lookup">
        <label>Description Details</label>
        <div class="lookup-result-text" id="lookup_${id}">Nhập số để tra cứu...</div>
      </div>
    </div>
    
    <label>Reality check / Thông số đo được</label>
    <input type="text" class="reality-check form-control" value="${card.realityCheck || ''}" placeholder="Nhập thông số thực tế...">
    <br>
    
    <label>Comments</label>
    <input type="text" class="comments form-control" value="${card.comments || ''}" placeholder="Nhập ghi chú hoặc ý kiến...">
    <br>
    
    <label>Pictures (Chỉ tải lên ảnh mới bổ sung)</label>
    <div class="photos-container-list"></div>
    <div class="preview-container-gallery" style="margin-top: 10px; display: flex; flex-wrap: wrap; gap: 15px;"></div>
  </div>
  `;
  container.insertAdjacentHTML("beforeend", html);
  
  if (card.imageUrls && card.imageUrls.length > 0) {
    const gallery = document.querySelector('[data-id="' + id + '"] .preview-container-gallery');
    card.imageUrls.forEach((url, index) => {
      if (!url.trim()) return;
      const wrapper = document.createElement("div");
      wrapper.className = "preview-wrapper existing-image";
      wrapper.style.position = "relative";
      wrapper.style.display = "inline-block";
      
      const link = document.createElement("a");
      link.href = url.trim().replace("thumbnail?sz=w500&id=", "open?id=");
      link.target = "_blank";
      
      const imgElement = document.createElement("img");
      imgElement.className = "preview-img-new";
      imgElement.src = url.trim();
      imgElement.style.width = "150px";
      imgElement.style.height = "150px";
      imgElement.style.objectFit = "cover";
      imgElement.style.borderRadius = "8px";
      imgElement.style.border = "3px solid #1a73e8";
      
      link.appendChild(imgElement);
      wrapper.appendChild(link);
      gallery.appendChild(wrapper);
    });
  }

  appendNewFileInputBox(id);

  if (!isNew && card.number) {
    const inputEl = document.querySelector('[data-id="' + id + '"] .number');
    handleNumberLookup(inputEl, id);
  }
}

function handleNumberLookup(inputElement, cardId) {
  const value = inputElement.value.trim();
  const resultDisplay = document.getElementById("lookup_" + cardId);
  
  if (!value) {
    resultDisplay.textContent = "Nhập số để tra cứu...";
    return;
  }
  
  const selectedLocations = [];
  document.querySelectorAll("#locOptions input:checked").forEach(cb => { selectedLocations.push(cb.value); });
  const currentLocation = selectedLocations.length > 0 ? selectedLocations[0] : "";
  
  if (!currentLocation) {
    resultDisplay.textContent = "⚠️ Hãy tích chọn Room (Location) phía trên trước!";
    return;
  }
  
  resultDisplay.textContent = "🔍 Đang tìm...";
  
  google.script.run.withSuccessHandler(function(responseText) {
    resultDisplay.textContent = responseText || "⚠️ Trống dữ liệu";
  }).lookupDescriptionByNumber(value, currentLocation);
}

function appendNewFileInputBox(cardId) {
  const cardDiv = document.querySelector('[data-id="' + cardId + '"]');
  const container = cardDiv.querySelector(".photos-container-list");
  const inputId = "img_inp_" + Date.now() + "_" + Math.floor(Math.random() * 1000);
  
  const div = document.createElement("div");
  div.className = "photo-input-item";
  div.id = "box_" + inputId;
  div.innerHTML = '<input type="file" id="' + inputId + '" class="photos form-control" accept="image/*" style="max-width: 180px; overflow: hidden;">';
  container.appendChild(div);

  const fileInput = div.querySelector("input");
  fileInput.addEventListener("change", function() {
    if (this.files && this.files[0]) {
      createSingleImagePreview(cardId, this.files[0], div.id);
      div.style.display = "none";
      appendNewFileInputBox(cardId);
    }
  });
}

function createSingleImagePreview(cardId, fileObj, sourceBoxId) {
  const cardDiv = document.querySelector('[data-id="' + cardId + '"]');
  const gallery = cardDiv.querySelector(".preview-container-gallery");
  const wrapper = document.createElement("div");
  wrapper.className = "preview-wrapper-new new-image";
  wrapper.setAttribute("data-sourcebox", sourceBoxId);
  wrapper.style.position = "relative";
  wrapper.style.display = "inline-block";
  
  const img = document.createElement("img");
  img.className = "preview-img-new";
  img.src = URL.createObjectURL(fileObj);
  img.style.width = "150px";
  img.style.height = "150px";
  img.style.objectFit = "cover";
  img.style.borderRadius = "4px";
  img.style.border = "1px solid #ddd";
  
  const removeBtn = document.createElement("button");
  removeBtn.type = "button";
  removeBtn.className = "btn-remove-img";
  removeBtn.innerHTML = "&times;";
  
  removeBtn.addEventListener("click", function() {
    wrapper.remove();
    const originalInputBox = document.getElementById(sourceBoxId);
    if (originalInputBox) originalInputBox.remove();
  });
  
  wrapper.appendChild(img);
  wrapper.appendChild(removeBtn);
  gallery.appendChild(wrapper);
}

function addCard() {
  createCard({ number: "", realityCheck: "", comments: "" });
}

async function submitForm() {
  const cards = document.querySelectorAll(".card");
  const saveCards = [];
  
  for (const card of cards) {
    const number = card.querySelector(".number").value.trim();
    if (!number) continue;
    
    const realityCheck = card.querySelector(".reality-check").value.trim();
    const comments = card.querySelector(".comments").value.trim();

    const fileInputs = card.querySelectorAll(".photo-input-item input[type='file']");
    const activeFiles = [];
    fileInputs.forEach(inp => {
      if (inp.files && inp.files[0]) activeFiles.push(inp.files[0]);
    });

    const matchedOrig = originalCards.find(c => c.number.trim() === number);
    if (matchedOrig) {
      const isRealityEqual = (matchedOrig.realityCheck || "") === realityCheck;
      const isCommentsEqual = (matchedOrig.comments || "") === comments;
      
      if (isRealityEqual && isCommentsEqual && activeFiles.length === 0) {
        continue; 
      }
    }

    let images = [];
    for (const file of activeFiles) {
      const base64 = await fileToBase64(file);
      images.push(base64);
    }
    
    saveCards.push({ number: number, realityCheck: realityCheck, comments: comments, images: images });
  }
  
  if (saveCards.length === 0) {
    alert("ℹ️ Không có dữ liệu nào thay đổi để cập nhật!");
    return;
  }
  
  const currentSelectedLocs = [];
  document.querySelectorAll("#locOptions input:checked").forEach(cb => { currentSelectedLocs.push(cb.value); });
  let finalLocation = currentSelectedLocs.join(", ");
  
  const payload = { location: finalLocation || "", cards: saveCards };
  
  google.script.run.withSuccessHandler(function(result) {
    if (result && result.success) {
      const toast = document.getElementById("toastMessage");
      toast.textContent = "🎉 " + result.message;
      toast.style.display = "block";
      
      setTimeout(function() {
        toast.style.display = "none";
        
        document.querySelectorAll("#locOptions input").forEach(cb => {
          cb.checked = currentSelectedLocs.includes(cb.value);
        });
        updateTriggerText(document.getElementById("locTrigger"), document.getElementById("locOptions"), "Select Location");
        
        document.getElementById("cardsContainer").innerHTML = "";
        originalCards = [];
        loadCardsFromServer(); 
      }, 2000);
    } else { alert("⚠️ Có lỗi xảy ra: " + (result && result.message ? result.message : "Thử lại")); }
  }).saveCards(payload);
}

function fileToBase64(file) {
  return new Promise(resolve => {
    const reader = new FileReader();
    reader.onload = e => resolve(e.target.result);
    reader.readAsDataURL(file);
  });
}
